\begin{frame}

    $\#include<queue>$
    \begin{itemize}
        \item push(element) : 큐에 원소를 추가(뒤에)
        \item pop() : 큐에 있는 원소를 삭제(앞에) 조회
        \item front() : 큐 제일 앞에 있는 원소를 반환
        \item back() : 큐 제일 뒤에 있는 원소를 반환
    \end{itemize}
\end{frame}
